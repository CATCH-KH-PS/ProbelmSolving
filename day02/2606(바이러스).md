```c++
#define _CRT_SECURE_NO_WARNINGS
#include <string>
#include <vector>
#include <queue>
#include <memory.h>
#include <algorithm>
#include <iostream>
#include <stack>
#include <unordered_set>
#include <math.h>
#include <utility>

using namespace std;

int g_n, g_m;
vector<int> g_v[100];

int check[100] = { 0 };
int answer = 0;

void go_find(int n)
{
	if (n != 0)
		answer++;
	for (int i = 0; i < g_v[n].size();i++)
	{
		if (check[g_v[n][i]] == 0)
		{
			check[g_v[n][i]] = 1;
			go_find(g_v[n][i]);
		}
	}
}

int	main(void)
{
	cin.tie(NULL);
	cout.tie(NULL);
	ios_base::sync_with_stdio(false);
	freopen("input.txt", "r", stdin);

	cin >> g_n >> g_m;
	int n, m;
	{
		if (g_n == 0)
		{
			cout << "0";
			return 0;
		}
	}
	for (int i = 0; i < g_m; i++)
	{
		cin >> n >> m;
		n -= 1;
		m -= 1;
		g_v[n].push_back(m);
		g_v[m].push_back(n);
	}
	check[0] = 1;
	go_find(0);
	cout << answer;
	return (0);
}
```
